This is simple font increase and decrease jquery plugin to increase or decrease the font size when you click the symbol 
            POSSIBLE VALUES:-
                activeCSSClass : "bFontActive" // this is the class to create custom class for the active state
                targetElements: "p" // this property is used to target HTML text element to increase the font size
                lineHeightPercentage: 1.67 // this property is used to define the line-height in terms of percentage
                fontIncreaseMultiplier: 2 // this property is the multiplier to increase font-size 
            -------------------------------------------------------------------------------------------------------------

            DEFAULT VALUES:-
                targetElements: "body p",
                fontIncreaseMultiplier: 2,
                lineHeightPercentage: 1.67,
                activeCSSClass: "bFontActive"
